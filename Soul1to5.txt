# Caesar Cipher Encryption Function

def encrypt(text, s):
    result = ""

    # Traverse through each character in the text
    for i in range(len(text)):
        char = text[i]

        # Encrypt uppercase characters
        if char.isupper():
            result += chr((ord(char) + s - 65) % 26 + 65)

        # Encrypt lowercase characters
        else:
            result += chr((ord(char) + s - 97) % 26 + 97)

    return result

# Check the above function
text = "HarshMhatre"
s = 3

print("Text   :", text)
print("Shift  :", s)
print("Cipher :", encrypt(text, s))

Vernam Cipher

# Function to clean text (remove spaces, keep only letters, convert to uppercase)
def clean(t):
    return "".join(c for c in t.upper() if c.isalpha())


# Vernam Cipher Encryption
def vernam_encrypt(pt, key):
    pt = clean(pt)
    key = clean(key)

    # Key length must match plaintext length
    if len(pt) != len(key):
        raise ValueError("Key length must equal plaintext length")

    # Encryption process
    return "".join(
        chr((ord(p) + ord(k) - 130) % 26 + 65)
        for p, k in zip(pt, key)
    )


# Vernam Cipher Decryption
def vernam_decrypt(ct, key):
    ct = clean(ct)
    key = clean(key)

    # Decryption process
    return "".join(
        chr((ord(c) - ord(k) + 26) % 26 + 65)
        for c, k in zip(ct, key)
    )


# -------- Test --------
pt = "HELLO"
key = "XMCKL"

ct = vernam_encrypt(pt, key)

print("Plaintext :", pt)
print("Key       :", key)
print("Ciphertext:", ct)
print("Decrypted :", vernam_decrypt(ct, key))

Q4 Aim:Write program to encrypt and decrypt strings using.
- DES Algorithm
# Install library first (run once in terminal)
# pip install pycryptodome

from Crypto.Cipher import DES
import base64


# Padding function (DES requires multiples of 8 bytes)
def pad(text):
    return text + (8 - len(text) % 8) * ' '


# Remove padding
def unpad(text):
    return text.rstrip()


# DES Encryption
def des_encrypt(text, key):
    # Ensure key is exactly 8 bytes
    key = key.ljust(8)[:8].encode()

    # Create DES cipher (ECB mode)
    cipher = DES.new(key, DES.MODE_ECB)

    # Encrypt and encode using Base64
    encrypted_text = cipher.encrypt(pad(text).encode())
    return base64.b64encode(encrypted_text).decode()


# DES Decryption
def des_decrypt(ciphertext, key):
    # Ensure key is exactly 8 bytes
    key = key.ljust(8)[:8].encode()

    # Create DES cipher
    cipher = DES.new(key, DES.MODE_ECB)

    # Decode Base64 and decrypt
    decrypted_text = cipher.decrypt(
        base64.b64decode(ciphertext)
    ).decode()

    return unpad(decrypted_text)


# -------- Test --------
text = "HELLO WORLD"
key = "secret12"

cipher = des_encrypt(text, key)
plain = des_decrypt(cipher, key)

print("\n--- DES Encryption & Decryption ---")
print("Plaintext  :", text)
print("Key        :", key)
print("Ciphertext :", cipher)
print("Decrypted  :", plain)
